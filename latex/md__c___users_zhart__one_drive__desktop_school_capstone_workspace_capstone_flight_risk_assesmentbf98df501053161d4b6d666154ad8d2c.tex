The {\ttfamily class\+Names} function takes any number of arguments which can be a string or object. The argument `\textquotesingle{}foo'{\ttfamily is short for}\{ foo\+: true \}\`{}. If the value associated with a given key is falsy, that key won\textquotesingle{}t be included in the output.


\begin{DoxyCode}{0}
\DoxyCodeLine{classNames('foo', 'bar'); // => 'foo bar'}
\DoxyCodeLine{classNames('foo', \{ bar: true \}); // => 'foo bar'}
\DoxyCodeLine{classNames(\{ 'foo-\/bar': true \}); // => 'foo-\/bar'}
\DoxyCodeLine{classNames(\{ 'foo-\/bar': false \}); // => ''}
\DoxyCodeLine{classNames(\{ foo: true \}, \{ bar: true \}); // => 'foo bar'}
\DoxyCodeLine{classNames(\{ foo: true, bar: true \}); // => 'foo bar'}
\DoxyCodeLine{}
\DoxyCodeLine{// lots of arguments of various types}
\DoxyCodeLine{classNames('foo', \{ bar: true, duck: false \}, 'baz', \{ quux: true \}); // => 'foo bar baz quux'}
\DoxyCodeLine{}
\DoxyCodeLine{// other falsy values are just ignored}
\DoxyCodeLine{classNames(null, false, 'bar', undefined, 0, 1, \{ baz: null \}, ''); // => 'bar 1'}

\end{DoxyCode}

This package is the official replacement for {\ttfamily class\+Set}, which was originally shipped in the React.\+js Addons bundle.

One of its primary use cases is to make dynamic and conditional {\ttfamily class\+Name} props simpler to work with (especially more so than conditional string manipulation). So where you may have the following code to generate a {\ttfamily class\+Name} prop for a {\ttfamily $<$button$>$} in React\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var Button = React.createClass(\{}
\DoxyCodeLine{  // ...}
\DoxyCodeLine{  render () \{}
\DoxyCodeLine{    var btnClass = 'btn';}
\DoxyCodeLine{    if (this.state.isPressed) btnClass += ' btn-\/pressed';}
\DoxyCodeLine{    else if (this.state.isHovered) btnClass += ' btn-\/over';}
\DoxyCodeLine{    return <button className=\{btnClass\}>\{this.props.label\}</button>;}
\DoxyCodeLine{  \}}
\DoxyCodeLine{\});}

\end{DoxyCode}


You can express the conditional classes more simply as an object\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var classNames = require('classnames');}
\DoxyCodeLine{}
\DoxyCodeLine{var Button = React.createClass(\{}
\DoxyCodeLine{  // ...}
\DoxyCodeLine{  render () \{}
\DoxyCodeLine{    var btnClass = classNames(\{}
\DoxyCodeLine{      btn: true,}
\DoxyCodeLine{      'btn-\/pressed': this.state.isPressed,}
\DoxyCodeLine{      'btn-\/over': !this.state.isPressed \&\& this.state.isHovered}
\DoxyCodeLine{    \});}
\DoxyCodeLine{    return <button className=\{btnClass\}>\{this.props.label\}</button>;}
\DoxyCodeLine{  \}}
\DoxyCodeLine{\});}

\end{DoxyCode}

There is an alternate version of {\ttfamily class\+Names} available which correctly dedupes classes and ensures that falsy classes specified in later arguments are excluded from the result set.

This version is slower (about 5x) so it is offered as an opt-\/in.

To use the dedupe version with Node.\+js, Browserify, or webpack\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{var classNames = require('classnames/dedupe');}
\DoxyCodeLine{}
\DoxyCodeLine{classNames('foo', 'foo', 'bar'); // => 'foo bar'}
\DoxyCodeLine{classNames('foo', \{ foo: false, bar: true \}); // => 'bar'}

\end{DoxyCode}
